```js
export function buildUpgradeRailModel({
  upgrades,
  factories,
  qsos,
  factoryCounts,
  purchasedUpgrades,
  upgradePurchaseMeta,
}) {
  const factoryIndex = new Map(factories.map((f, i) => [f.id, i]))
  const upgradeIndex = new Map(upgrades.map((u, i) => [u.id, i]))
  const safeMeta = upgradePurchaseMeta || {}

  const normalized = upgrades.map(upgrade => {
    const owned = factoryCounts[upgrade.factoryId] || 0
    const isPurchased = purchasedUpgrades.has(upgrade.id)
    const isAvailable = !isPurchased && owned >= upgrade.threshold
    const isAffordable = isAvailable && qsos >= upgrade.baseCost
    const costDelta = isAvailable && !isAffordable ? upgrade.baseCost - qsos : 0n
    const rawPurchasedAt = Number(safeMeta[upgrade.id])
    const purchasedAt = Number.isFinite(rawPurchasedAt) && rawPurchasedAt > 0 ? rawPurchasedAt : 0

    return {
      ...upgrade,
      isPurchased,
      isAvailable,
      isAffordable,
      costDelta,
      purchasedAt,
      factorySortIndex: factoryIndex.get(upgrade.factoryId) ?? Number.MAX_SAFE_INTEGER,
      upgradeSortIndex: upgradeIndex.get(upgrade.id) ?? Number.MAX_SAFE_INTEGER,
    }
  })

  const byStableOrder = (a, b) => {
    if (a.baseCost !== b.baseCost) return a.baseCost < b.baseCost ? -1 : 1
    if (a.factorySortIndex !== b.factorySortIndex) return a.factorySortIndex - b.factorySortIndex
    return a.upgradeSortIndex - b.upgradeSortIndex
  }

  const readyToBuy = normalized.filter(u => u.isAvailable && u.isAffordable).sort(byStableOrder)
  const almostThere = normalized
    .filter(u => u.isAvailable && !u.isAffordable)
    .sort((a, b) => {
      if (a.costDelta !== b.costDelta) return a.costDelta < b.costDelta ? -1 : 1
      return byStableOrder(a, b)
    })
  const recentlyPurchased = normalized
    .filter(u => u.isPurchased)
    .sort((a, b) => {
      if (a.purchasedAt !== b.purchasedAt) return b.purchasedAt - a.purchasedAt
      return byStableOrder(a, b)
    })
  const lockedByThreshold = normalized.filter(u => !u.isAvailable && !u.isPurchased).sort(byStableOrder)

  const priorityRow = [...readyToBuy, ...almostThere, ...recentlyPurchased].slice(0, 5)

  return {
    readyToBuy,
    almostThere,
    recentlyPurchased,
    lockedByThreshold,
    priorityRow,
    priorityRowIds: priorityRow.map(u => u.id),
  }
}
```

const upgradePurchaseMeta = ref({})

function normalizeUpgradePurchaseMeta(meta = {}) {
  const normalized = {}
  for (const [id, value] of Object.entries(meta || {})) {
    const n = Number(value)
    normalized[id] = Number.isFinite(n) && n > 0 ? n : 0
  }
  return normalized
}

function buyUpgrade(upgradeId) {
  const upgrade = UPGRADES.find(u => u.id === upgradeId)
  if (!upgrade) return false
  if (purchasedUpgrades.value.has(upgradeId)) return false
  const ownedCount = factoryCounts.value[upgrade.factoryId] || 0
  if (ownedCount < upgrade.threshold) return false
  if (qsos.value < upgrade.baseCost) return false

  qsos.value -= upgrade.baseCost
  purchasedUpgrades.value.add(upgradeId)
  upgradePurchaseMeta.value[upgradeId] = Date.now()
  return true
}

// save payload
upgradePurchaseMeta: upgradePurchaseMeta.value,

// load
upgradePurchaseMeta.value = normalizeUpgradePurchaseMeta(state.upgradePurchaseMeta)

// reset paths
upgradePurchaseMeta.value = {}
